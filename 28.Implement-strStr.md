/**
 * @param {string} haystack
 * @param {string} needle
 * @return {number}
 */
var strStr = function(haystack, needle) {
    if (needle === '') return 0;
    var index = -1;
    for (var i = 0; i < haystack.length - needle.length + 1; i++) {
        if (haystack[i] === needle[0]) {
            index = i;
            for (var j = 1; j < needle.length; j++) {
                if (haystack[i + j] !== needle[j]) {
                    index = -1;
                    break;
                }
            }
        }
        if (index !== -1) {
            return index;
        }
    }
    return index;
};
